💬 Keluhan CS

Laporan kendala dari pelanggan
⚠️ Tabel complaints belum ada atau RLS belum diset di Supabase.
Jalankan SQL berikut di Supabase → SQL Editor:
CREATE TABLE IF NOT EXISTS complaints (
  id TEXT PRIMARY KEY,
  username TEXT,
  order_id TEXT,
  kategori TEXT,
  pesan TEXT,
  kontak TEXT,
  status TEXT DEFAULT 'open',
  reply TEXT,
  reply_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT now()
);
ALTER TABLE complaints ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon_insert" ON complaints FOR INSERT TO anon WITH CHECK (true);
CREATE POLICY "anon_select" ON complaints FOR SELECT TO anon USING (true);
CREATE POLICY "anon_update" ON complaints FOR UPDATE TO anon USING (true);
